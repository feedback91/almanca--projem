(Diğer tüm fiiller buraya eklenebilir) ] st.session_state.skor = 0 st.session_state.toplam = 0 # --- KENAR ÇUBUĞU (Sidebar) --- st.sidebar.title("📚 Eğitim Menüsü") mod = st.sidebar.radio("Bir Mod Seç:", ["Pratik Test", "Tüm Liste", "Tentenistant Kodu"]) # --- MOD 1: PRATİK TEST --- if mod == "Pratik Test": st.title("🎯 Fiil + Edat Pratiği") # Rastgele soru seçimi soru = random.choice(st.session_state.fiiller) st.subheader(f"'{soru['f']}'") st.write(f"Anlamı: *{soru['t']}*") secenekler = ["an", "auf", "für", "von", "mit", "um", "über", "in", "nach", "vor"] # Doğru cevabı ve 3 rastgele yanlışı al dogru = soru['p'] yanlislar = [p for p in secenekler if p != dogru] siklar = random.sample(yanlislar, 3) + [dogru] random.shuffle(siklar) cevap = st.radio("Doğru edatı seçin:", siklar, key="quiz") if st.button("Kontrol Et"): st.session_state.toplam += 1 if cevap == dogru: st.success(f"Harika!
